rho=np.load('Na_ae_density.npy') # all-electron density from GPAW using LDA
V=np.load('Na_ae_potential.npy') # all-electron potential from GPAW using LDA
atoms=read('Na.cif',index=0) # atomic configuration
V=-V/factor

print('density shape:',rho.shape,' potential shape:',V.shape)
Lx,Ly,Lz=np.diag(atoms.get_cell())
Nx,Ny,Nz=V.shape

# DFT potential directly using paw corrections
V_dft_Na=create_potential_slices(V-np.min(V),10,(Lx,Ly,Lz))

# DFT potential from charge density
V=poisson_solver(rho,atoms,smooth=0,units='QSTEM')
V_charge_Na=create_potential_slices(V,8,(Lx,Ly,Lz))

# QSTEM potential
qstem=PyQSTEM('TEM')
qstem.set_atoms(atoms)
qstem.build_wave('plane',300,(Nx,Ny))
qstem.build_potential(10)
V_qstem_Na=qstem.get_potential_or_transfunc()
